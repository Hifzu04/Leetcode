class Solution {
public:
    
    bool f(long long n,long long i){
         if(i==n){
            return true ;
         }
         if(i>n) return false ;

         return f(n,i*2);
    }


    bool isPowerOfTwo(int n) {
        long long i =1;
        //if(n<0) n= (n*-1);
       // if(n%2==0) n=n/2;
        return f(n , i);
    }
};
